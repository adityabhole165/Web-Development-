let upperCaseOne = (id) => {
    setTasks((prevTasks) =>
        prevTasks.map(
            (todo) => {
                if(todo.id == id) {
                    return{
                        ...todo,
                        task:todo.task.toUppercase(),
                    };
                }else {
                    return todo;
                }
            }
        )
    );
};
